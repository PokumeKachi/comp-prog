vector<int> g[N];
int depth[N], parent[N];
map<pair<int,int>, int> cnt;
map<int,int> tot;

bfs(1){
  queue<int> q; q.push(1);
  depth[1]=0; parent[1]=0;
  while(!q.empty()){
    int u=q.front(); q.pop();
    for(int v:g[u]) if(v!=parent[u]){
      parent[v]=u;
      depth[v]=depth[u]+1;
      q.push(v);
    }
  }
}

for(v=1..n){
  cnt[{depth[v], parent[v]}]++;
  tot[depth[v]]++;
}

ans=0;
for(each depth d){
  mx = max cnt[(d,*)];
  ans = max(ans, max(mx, tot[d]));
}
